#include<iostream>
#include<string>
using namespace std;
// 
int main(){
    int rows,cols;
    cout<<"Size: ";
    cin>>rows>>cols;
    int **ptr=new int*[rows];
    for(int i=0;i<rows;i++){
        ptr[i]=new int[cols];
    }
    for(int r=0;r<rows;r++){
        for(int c=0;c<cols;c++){
            ptr[r][c]=c;
        }
    }
    cout<<"Array: "<<endl;
    for(int r=0;r<rows;r++){
        for(int c=0;c<cols;c++){
            cout<<ptr[r][c];
        }
        cout<<endl;
    }
    for(int i=0;i<rows;i++){
        delete[] ptr[i];
    }
    delete[] ptr;
    ptr=NULL;
    return 0;
}
