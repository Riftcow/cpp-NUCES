#include<iostream>
#include<string>
using namespace std;
class demo{
    int a,b;
    public:
    demo(){
        a=0;
        b=0;
    }
    demo(int x,int y){
        a=x;
        b=y;
    }
    demo(demo &obj){
        cout<<"copy constructor"<<endl;
        a=obj.a;
        b=obj.b;
    }
    void show(){
        cout<<"a: "<<a<<endl;
        cout<<"b: "<<b<<endl;
        cout<<"<--------------------------------->"<<endl;
    }
};
int main(){
    demo d(4,5);
    demo d1=d;//d1(d); d1.a=d.a,d1.b=d.b
    d.show();
    d1.show();
    return 0;
}
